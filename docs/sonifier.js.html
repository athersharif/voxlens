/**
 * @namespace sonifier
 */

import { sonificationSettings } from './settings';
import { toFrequency } from './utils';

/**
 * Generates the sonified respoonse.
 * @memberOf sonifier
 * @param {Object} Tone - The instance from tonejs library.
 * @param {Object} data - The data from the viz.
 * @param {string[]} data.x - Values of the independent variable.
 * @param {string[]} data.y - Values of the dependent variable.
 * @returns {Object[]} - Oscillations in the sonified response.
 */
const sonifier = (Tone, data) => {
  const sortedData = [...data.y].sort((a, b) => a - b);
  const maxValue = sortedData[sortedData.length - 1];
  const minValue = sortedData[0];

  let oscillations = [];
  let start = 0;
  let stop = start + sonificationSettings.oscillator.interval;

  data.y
    .map((d) =>
      toFrequency(
        d,
        minValue,
        maxValue,
        sonificationSettings.frequency.minimum,
        sonificationSettings.frequency.maximum
      )
    )
    .forEach((d) => {
      const osc = new Tone.OmniOscillator(
        sonificationSettings.note,
        sonificationSettings.oscillator.type
      ).toDestination();
      osc.frequency.value = d;
      osc.volume.value = sonificationSettings.volume;
      osc.onstop = () => osc.dispose();
      osc.sync().start(start).stop(stop);

      oscillations.push(osc);
      start = stop;
      stop = stop + sonificationSettings.oscillator.interval;
    });

  console.log('[VoxLens] Playing sonification');

  Tone.Transport.start();

  return oscillations;
};

export default sonifier;
